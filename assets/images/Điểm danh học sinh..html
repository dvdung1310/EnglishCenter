function diemDanh() {
    var checkboxes = document.querySelectorAll('#attendance tbody input[type="checkbox"]');
    var attendanceData = [];

    checkboxes.forEach(function(checkbox, index) {
        var maHS = ds_hocsinh[index].MaHS;
        var tenHS = ds_hocsinh[index].TenHS;
        var diemDanh = checkbox.checked;

        attendanceData.push({
            maHS: maHS,
            tenHS: tenHS,
            diemDanh: diemDanh
        });
    });

    console.log(attendanceData);

    // Lấy biểu mẫu điểm danh
    var form = document.getElementById('attendanceForm');

    // Gán dữ liệu điểm danh vào một trường ẩn trong biểu mẫu
    var input = document.createElement('input');
    input.setAttribute('type', 'hidden');
    input.setAttribute('name', 'attendanceData');
    input.setAttribute('value', JSON.stringify(attendanceData));
    form.appendChild(input);

    // Gửi biểu mẫu khi nhấn nút submit
    form.addEventListener('submit', function(event) {
        event.preventDefault(); // Ngăn chặn gửi lại trang
        var formData = new FormData(form);

        // Gửi dữ liệu điểm danh đến trang xử lý server
        fetch('/update_attendance.php', {
                method: 'POST',
                body: formData
            })
            .then(function(response) {
                if (response.ok) {
                    // Điểm danh thành công
                    alert('Điểm danh thành công!');
                } else {
                    // Xử lý lỗi response
                    alert('Điểm danh thất bại.');
                }
            })
            .catch(function(error) {
                // Xử lý lỗi fetch
                alert('Đã xảy ra lỗi khi gửi yêu cầu điểm danh.');
                console.error(error);
            });
    });

    // Submit biểu mẫu khi nhấn nút submit
    form.submit();
}

Khi dữ liệu điểm danh được gửi đến trang update_attendance.php và bạn đã xử lý nó, bạn có thể hiển thị dữ liệu đó theo ý muốn. Dưới đây là một ví dụ cách hiển thị dữ liệu điểm danh trong PHP:

php
<?php
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    // Lấy dữ liệu điểm danh từ biểu mẫu

    // Hiển thị dữ liệu điểm danh

Dưới đây là một ví dụ cơ bản về cách lấy dữ liệu từ bảng điểm danh và gửi nó đến PHP bằng AJAX:

function diemDanh() {
  // Lấy danh sách các checkbox trong bảng điểm danh
  var checkboxes = document.querySelectorAll('#attendance tbody input[type="checkbox"]');
  
  // Tạo một mảng để lưu trữ dữ liệu điểm danh
  var danhSachDiemDanh = [];
  
  // Lặp qua từng checkbox và lấy dữ liệu tương ứng
  checkboxes.forEach(function(checkbox) {
    var stt = checkbox.parentElement.parentElement.cells[0].textContent;
    var maHS = checkbox.parentElement.parentElement.cells[1].textContent;
    var tenHS = checkbox.parentElement.parentElement.cells[2].textContent;
    var diemDanh = checkbox.checked ? 1 : 0;
    
    // Tạo một đối tượng để lưu trữ dữ liệu điểm danh
    var diemDanhObj = {
      stt: stt,
      maHS: maHS,
      tenHS: tenHS,
      diemDanh: diemDanh
    };
    
    // Thêm đối tượng vào mảng danhSachDiemDanh
    danhSachDiemDanh.push(diemDanhObj);
  });
  
  // Gửi dữ liệu danhSachDiemDanh đến PHP bằng AJAX
  var xhttp = new XMLHttpRequest();
  xhttp.onreadystatechange = function() {
    if (this.readyState === 4 && this.status === 200) {
      // Xử lý phản hồi từ PHP sau khi gửi dữ liệu thành công
      console.log(this.responseText);
    }
  };

  xhttp.open("POST", "your-php-file.php", true);
  xhttp.setRequestHeader("Content-type", "application/json");
  xhttp.send(JSON.stringify(danhSachDiemDanh));
}

Hàm session_start() trong PHP được sử dụng để khởi động một phiên làm việc (session) mới hoặc khôi phục một phiên làm việc đã tồn tại. Nó phải được gọi trước khi sử dụng bất kỳ biến phiên nào hoặc trước khi gửi dữ liệu session cho trình duyệt.

Dưới đây là cách sử dụng session_start() trong PHP:

php
<?php
session_start();

// Sử dụng biến session
$_SESSION['username'] = 'John';

// Lấy giá trị của biến session
$username = $_SESSION['username'];

// In ra giá trị của biến session
echo $username;
?>

Trong ví dụ trên, chúng ta đã khởi động một phiên làm việc mới bằng cách gọi session_start(). Sau đó, chúng ta đã sử dụng biến $_SESSION để lưu trữ dữ liệu session. Trong trường hợp này, chúng ta đã lưu trữ tên người dùng là "John" vào biến $_SESSION['username']. Sau đó, chúng ta đã lấy giá trị của biến session này và in ra màn hình.

Lưu ý rằng session_start() phải được gọi trước khi bất kỳ mã HTML hoặc mã PHP nào được xuất ra trình duyệt, và nó chỉ được gọi một lần duy nhất trong mỗi trang PHP.

Hơn nữa, để sử dụng biến session, bạn cần đảm bảo rằng session đã được khởi động (session_start()) ở tất cả các trang PHP liên quan và trước khi truy cập hoặc thao tác với các biến session.

Biến $_SESSION['username'] sẽ không còn giá trị khi một trong các điều kiện sau xảy ra:

  1. Khi bạn gọi hàm session_destroy(): Hàm này được sử dụng để hủy toàn bộ dữ liệu session và làm cho tất cả các biến session không còn khả dụng. Ví dụ: session_destroy();

  2. Khi bạn gọi hàm session_unset(): Hàm này được sử dụng để hủy toàn bộ dữ liệu của biến session, nhưng phiên làm việc session vẫn còn tồn tại. Ví dụ: session_unset();

  3. Khi session hết hạn: Một session có thể được đặt để tự động hết hạn sau một khoảng thời gian xác định. Khi hết hạn, các biến session sẽ không còn khả dụng. Thời gian hết hạn của session được cấu hình trong tệp php.ini bằng cách thiết lập giá trị cho session.gc_maxlifetime.

  4. Khi bạn đóng trình duyệt: Khi bạn đóng trình duyệt, thông thường phiên làm việc session sẽ không còn tồn tại trên máy chủ.

  5. Khi bạn xóa biến session cụ thể: Bạn có thể xóa một biến session cụ thể bằng cách sử dụng lệnh unset() trong PHP. Ví dụ: unset($_SESSION['username']);

Chú ý rằng giá trị của biến session có thể tồn tại cho đến khi một trong các điều kiện trên xảy ra.
